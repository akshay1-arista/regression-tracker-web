// makeRequest() helper with AbortController async makeRequest(key, url, options = {}) { if (this._pendingRequests.has(key)) { this._pendingRequests.get(key).abort(); // Cancel previous } const controller = new AbortController(); this._pendingRequests.set(key, controller); try { const response = await fetch(url, { ...options, signal: controller.signal }); this._pendingRequests.delete(key); return await response.json(); } catch (err) { this._pendingRequests.delete(key); if (err.name === 'AbortError') { return null; // Request cancelled } throw err; } }
